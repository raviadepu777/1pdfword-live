# Use a lean, official Node.js 18 image
FROM node:18-slim

# Install LibreOffice
RUN apt-get update && apt-get install -y \
    libreoffice \
    --no-install-recommends \
    && apt-get clean \
    && rm -rf /var/lib/apt/lists/*

# Set the working directory
WORKDIR /usr/src/app

# Copy package.json to install dependencies
COPY package*.json ./
RUN npm install

# Copy the rest of the application code (including server.js and index.html)
COPY . .

# Expose the port the app runs on
EXPOSE 3000

# The command to start the server
CMD [ "node", "server.js" ]
